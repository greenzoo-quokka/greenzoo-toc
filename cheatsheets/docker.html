# 도커 로그인

docker login -u [토큰] [주소] [비밀번호]
1

# 컨테이너 생성

Dockfile 있는 경로에서 실행

docker build -t [이미지이름] .
1

# 도커이미지 실행

Dockfile 있는 경로에서 실행

docker run -d -p [호스프포트넘버]:[컨테이너포트넘버] [이미지이름]
1

# 도커이미지 태그변경

docker image tag [원본이미지이름] [변경이미지이름]
1

# 도커이미지 Push

docker push [이미지이름]
1

# 도커이미지 Pull

docker pull [이미지이름]
